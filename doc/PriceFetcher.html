19 def self.call(product)
20   return product if product.source_url.blank?
21   return product unless product.auto_refresh?
22 
23   upgrade_lululemon_source_url!(product)
24 
25   result = PriceScrapers.fetch(product.source_url, timeout: 5)
26   apply_resolved_source_url!(product, result)
27 
28   if result.price.present?
29     last_scraped = product.price_records
30                           .where(source: "scraped")
31                           .order(recorded_at: :desc)
32                           .first
33     if last_scraped.nil? || last_scraped.price != result.price
34       product.price_records.create!(
35         price:       result.price,
36         store_name:  result.store_name,
37         url:         result.resolved_url.presence || product.source_url,
38         recorded_at: result.fetched_at,
39         source:      "scraped"
40       )
41     end
42   end
43 
44   updates = { last_fetched_at: Time.current, last_fetch_error: nil }
45   updates[:stock_status] = result.availability if result.availability.present?
46   product.update_columns(updates)
47   product
48 rescue PriceScrapers::Error => e
49   product.update_columns(
50     last_fetch_error: e.message.to_s.first(250),
51     last_fetched_at: Time.current
52   )
53   product
54 end

      Fetch the latest price for a single product.

      Behaviour:

      - Skips products with no source_url (manual-only products are untouched).
- Catches all PriceScrapers::Error so callers (controllers / scheduler)
  never crash. Failures are surfaced via product.last_fetch_error.
- DEDUP: only writes a new PriceRecord(source: "scraped") when the price
  actually differs from the last scraped record. This means even an
  hourly cron won't pollute price history with thousands of identical
  rows; the chart shows only real price changes.
- NEVER mutates name / image_url / description / category. Those are
  populated only at product creation time (in ProductsController#create).
- DOES refresh stock_status when the page exposes availability, since
  (like price) it changes over time. Left untouched when unknown so a
  page that omits availability doesn't wipe a previously known status.

      Returns the product (always), so callers can chain or inspect.

      Refresh every product that has a source_url. CLI / emergency use only —the cron path enqueues RefreshPricesJob instead (async + batched).

 60 def self.refresh_batch(limit:, min_age: RefreshSchedule.stale_after, sleep_between: 0)
 61   started_at = Time.current
 62   total = Product.refreshable.count
 63   catalog_with_url = Product.with_trackable_url.count
 64 
 65   scope = Product.refreshable
 66                  .where("last_fetched_at IS NULL OR last_fetched_at < ?", min_age.ago)
 67                  .order(Arel.sql("last_fetched_at ASC NULLS FIRST"))
 68                  .limit(limit)
 69 
 70   succeeded = failed = 0
 71   failures = []
 72   scope.includes(:user).find_each do |product|
 73     call(product)
 74     if product.last_fetch_error.present?
 75       failed += 1
 76       failures << failure_detail_for(product)
 77     else
 78       succeeded += 1
 79     end
 80     sleep sleep_between if sleep_between.positive?
 81   end
 82 
 83   stale_remaining = Product.refreshable
 84                            .where("last_fetched_at IS NULL OR last_fetched_at < ?", min_age.ago)
 85                            .count
 86 
 87   duration = (Time.current - started_at).round(1)
 88   summary = {
 89     total: total,
 90     catalog_with_url: catalog_with_url,
 91     batch_size: limit,
 92     runs_per_cycle: RefreshSchedule.runs_per_cycle,
 93     attempted: succeeded + failed,
 94     succeeded: succeeded,
 95     failed: failed,
 96     stale_remaining: stale_remaining,
 97     failures: failures,
 98     duration: duration

100   Rails.logger.info("[PriceFetcher] refresh_batch finished — #{summary.inspect}")
101   summary
102 end

      Refresh up to limit stale products, oldest-first. Used by RefreshPricesJob on each cron tick. Batch size is computed by RefreshSchedule from the current product count so the catalog is covered within the refresh window without redeploying when load grows.
