      module MailerSettings

      Shared Action Mailer configuration driven by ENV.

      Production (and development when testing) sends real mail when SMTP_* vars are set — typically SendGrid on Heroku:

      SMTP_ADDRESS=smtp.sendgrid.net
SMTP_USERNAME=apikey
SMTP_PASSWORD=<SendGrid API key>
MAILER_FROM="PriceTracker <you@verified-sender.com>"
APP_URL=https://smart-shoppinglist-....herokuapp.com

41 def mailer_host
42   return ENV["MAILER_HOST"] if ENV["MAILER_HOST"].present?
43   return nil if ENV["APP_URL"].blank?
44

46 rescue URI::InvalidURIError
